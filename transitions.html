#styles.css body { font-family: Arial, sans-serif; text-align: center; background-color: #f0f0f0; } #animatedBox { width: 100px; height: 100px; background-color: lightblue; margin: 20px auto; position: relative; transition: transform 0.5s ease-in-out; } .animate { transform: rotate(360deg) scale(1.5); } // Select elements const animateButton = document.getElementById("animateButton"); const animatedBox = document.getElementById("animatedBox"); const savePreferenceButton = document.getElementById("savePreference"); // Function to trigger animation function toggleAnimation() { animatedBox.classList.toggle("animate"); } // Store user preference in localStorage function saveAnimationPreference() { const isAnimated = animatedBox.classList.contains("animate"); localStorage.setItem("animationState", isAnimated); alert("Animation preference saved!"); } // Retrieve preference on page load window.onload = function () { const savedState = localStorage.getItem("animationState"); if (savedState === "true") { animatedBox.classList.add("animate"); } }; #script.jss // Attach event listeners animateButton.addEventListener("click", toggleAnimation); savePreferenceButton.addEventListener("click", saveAnimationPreference);
